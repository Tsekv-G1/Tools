📋 Google Apps Script — бүтэн doGet(e) код
Доорх бүтэн кодыг Google Apps Script дээр doGet(e) функц болгон хуулж, 2 Sheet үүсгэнэ үү:
Passwords (name, uid, plan, pw, expiry, days) · Registrations (phone, pw, plan, status, date)
function doGet(e) { var action = e.parameter.action || 'list'; var ss = SpreadsheetApp.getActiveSpreadsheet(); var CORS = { 'Access-Control-Allow-Origin': '*' }; // ── LIST passwords ────────────────────────────────── if (action === 'list' || !action) { var sheet = ss.getSheetByName('Passwords'); var data = sheet.getDataRange().getValues(); var rows = []; for (var i = 1; i < data.length; i++) { rows.push({ name:data[i][0], uid:data[i][1], plan:data[i][2], pw:data[i][3], expiry:data[i][4], days:data[i][5] }); } return out(rows); } // ── ADD password ──────────────────────────────────── if (action === 'add') { var sh = ss.getSheetByName('Passwords'); sh.appendRow([e.parameter.name, e.parameter.uid, e.parameter.plan, e.parameter.pw, parseInt(e.parameter.expiry), parseFloat(e.parameter.days)]); return out({ ok: true }); } // ── DELETE password ───────────────────────────────── if (action === 'delete') { var sh = ss.getSheetByName('Passwords'); var d = sh.getDataRange().getValues(); for (var i = 1; i < d.length; i++) { if (d[i][3] === e.parameter.pw) { sh.deleteRow(i+1); break; } } return out({ ok: true }); } // ── REGISTER (landing.html-аас) ───────────────────── if (action === 'register') { var sh = ss.getSheetByName('Registrations'); if (!sh) sh = ss.insertSheet('Registrations'); // Давхардал шалгах var d = sh.getDataRange().getValues(); for (var i = 1; i < d.length; i++) { if (d[i][0] == e.parameter.phone) { sh.getRange(i+1, 4).setValue('PENDING'); return out({ ok: true, updated: true }); } } sh.appendRow([e.parameter.phone, e.parameter.pw, e.parameter.plan || 'pro', e.parameter.status || 'PENDING', e.parameter.date || new Date().toLocaleString()]); return out({ ok: true }); } // ── LIST registrations (admin) ─────────────────────── if (action === 'listRegistrations') { var sh = ss.getSheetByName('Registrations'); if (!sh) return out([]); var d = sh.getDataRange().getValues(); var rows = []; for (var i = 1; i < d.length; i++) { rows.push({ phone:d[i][0], pw:d[i][1], plan:d[i][2], status:d[i][3], date:d[i][4] }); } return out(rows); } // ── APPROVE registration ───────────────────────────── if (action === 'approveReg') { var sh = ss.getSheetByName('Registrations'); if (!sh) return out({ ok:false }); var d = sh.getDataRange().getValues(); for (var i = 1; i < d.length; i++) { if (d[i][0] == e.parameter.phone) { sh.getRange(i+1, 4).setValue('APPROVED'); return out({ ok: true }); } } return out({ ok: false, msg: 'not found' }); } // ── DELETE registration ────────────────────────────── if (action === 'deleteReg') { var sh = ss.getSheetByName('Registrations'); if (!sh) return out({ ok:false }); var d = sh.getDataRange().getValues(); for (var i = 1; i < d.length; i++) { if (d[i][0] == e.parameter.phone) { sh.deleteRow(i+1); break; } } return out({ ok: true }); } return out({ error: 'unknown action' }); } function out(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
